function generateJSON() { const tagsInput = document.getElementById('tags').value; const tagsArray = tagsInput.split(',').map(tag => tag.trim()).filter(tag => tag !== ""); const modObject = { id: document.getElementById('id').value, title: document.getElementById('title').value, category: document.getElementById('category').value, brand: document.getElementById('category').value === 'cars' ? prompt("نام برند ماشین را وارد کنید (مثال: BMW):") : "", image: document.getElementById('image').value, size: document.getElementById('size').value, version: document.getElementById('version').value, tags: tagsArray, download: document.getElementById('download').value, description: prompt("توضیحات متنی مود را جهت نمایش در پاپ‌آپ وارد کنید:"), author: { name: prompt("نام خود را به عنوان نویسنده وارد کنید:"), avatar: `https://api.dicebear.com/7.x/bottts/svg?seed=${Math.random()}` } }; document.getElementById('jsonOutput').value = ",\n" + JSON.stringify(modObject, null, 2); }
